Функция ВСтроку(Ст) Экспорт
	
	Стр = "{";
	Первое = Истина;
	
	Для Каждого КлючЗначение Из Ст Цикл
		Если Не Первое Тогда
			Стр = Стр + ",";
		Иначе
			Первое = Ложь;
		КонецЕсли;
		Стр = Стр + "'" + КлючЗначение.Ключ + "':'" + КлючЗначение.Значение + "'";
	КонецЦикла;
	
	Возврат Стр + "}";
